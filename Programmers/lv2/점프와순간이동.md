처음 dp 재귀방식으로 풀이 -> 답은 맞으나 효율성 테스트 실패.


```java
import java.util.*;

public class Solution {
    
    public static int[] D;
    public int solution(int n) {
        int ans = 0;
        D = new int[n + 1];
        if(n == 1) {
            return 1;
        }

        D[1] = 1;
        for (int i = 2; i <= n; i++) {
            D[i] = Integer.MAX_VALUE;
        }
        
        return dp(n);
    }
    
    public static int dp(int n) {
        if (D[n] != Integer.MAX_VALUE) {
            return D[n];
        }
        
        if(n % 2 == 0) {
            D[n] = Math.min(dp(n-1) + 1, dp(n / 2));
        } else {
            D[n] = dp(n-1) + 1;
        }
        
        return D[n];
    }
}
```
